实现 resolve 可以这样

```js
{
    var resolve = function(value){
        return function(fn){
            fn(value) // 此处 fn 就是 then 的回调函数
        }
    }
    let _thenFn = resolve(value)
    this.then = function(fn){
        _thenFn(fn)
    }
}
```
